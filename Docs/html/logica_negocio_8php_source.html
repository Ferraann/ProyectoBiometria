40 $sql = "
41 SELECT tm.medida, tm.unidad, AVG(m.valor) AS promedio
42 FROM medicion m
43 INNER JOIN tipo_medicion tm ON m.tipo_medicion_id = tm.id
44 WHERE
45 CAST(SUBSTRING_INDEX(m.localizacion, ',', 1) AS DECIMAL(10,6)) BETWEEN ? AND ?
46 AND CAST(SUBSTRING_INDEX(m.localizacion, ',', -1) AS DECIMAL(10,6)) BETWEEN ? AND ?
47 GROUP BY tm.id, tm.medida, tm.unidad
48 ORDER BY tm.medida ASC
49 ";

51 $stmt = $conn->prepare($sql);
52 $stmt->bind_param("dddd", $lat_min, $lat_max, $lon_min, $lon_max);

55
56 $promedios = [];

59 "medida" => $row['medida'],

63 }

66 "status" => "ok",
67 "rango" => [
68 "lat_min" => $lat_min,
69 "lat_max" => $lat_max,

83 return ["status" => "error", "mensaje" => "Falta el ID del usuario"];

85
86 $id = $data['id'];

88 // Obtenemos los valores enviados (si no vienen, no se modifican)

91 $correo = $data['gmail'] ?? null;

94 // Construimos SQL dinámico según lo que llegó
95 $campos = [];

100 $campos[] = "nombre = ?";

106 $campos[] = "apellidos = ?";
107 $params[] = $apellidos;
108 $tipos .= "s";

112 $campos[] = "gmail = ?";
113 $params[] = $correo;
114 $tipos .= "s";
115 }
116
117 if ($password !== null) {
118 $hash = password_hash($password, PASSWORD_DEFAULT);
119 $campos[] = "password = ?";
120 $params[] = $hash;
121 $tipos .= "s";
122 }

124 if (empty($campos)) {
125 return ["status" => "error", "mensaje" => "No hay datos para actualizar"];
126 }
127
128 // SQL final
129 $sql = "UPDATE usuario SET " . implode(", ", $campos) . " WHERE id = ?";
130 $params[] = $id;
131 $tipos .= "i";
132
133 $stmt = $conn->prepare($sql);
134 $stmt->bind_param($tipos, ...$params);
135
136 if (!$stmt->execute()) {
137 return ["status" => "error", "mensaje" => $conn->error];
138 }

140 // -------------------------------

160function obtenerIncidenciasActivas($conn)
161{
162 $sql = "
163 SELECT
164 i.id,
165 u.nombre AS usuario,
166 i.titulo,
167 i.descripcion,
168 i.fecha_creacion,
169 e.nombre AS estado,
170 i.id_tecnico,
171 COALESCE(tu.nombre, 'Sin asignar') AS tecnico
172 FROM incidencias i
173 LEFT JOIN usuario u ON i.id_user = u.id
174 LEFT JOIN estado_incidencia e ON i.estado_id = e.id
175 LEFT JOIN tecnicos t ON i.id_tecnico = t.usuario_id
176 LEFT JOIN usuario tu ON t.usuario_id = tu.id
177 WHERE e.nombre NOT IN ('Cerrada', 'Cancelada')
178 ORDER BY i.fecha_creacion DESC
179";
180
181 $result = $conn->query($sql);
182 $incidencias = [];
183 while ($row = $result->fetch_assoc()) {
184 $incidencias[] = $row;
185 }
186 return $incidencias;
