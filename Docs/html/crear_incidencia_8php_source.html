8// Función para crear una nueva incidencia en el sistema.

10// de forma dinámica e inserta el registro en la base de datos.

12
13function crearIncidencia($conn, $data)
14{
15 if (!isset($data['id_user'], $data['titulo'], $data['descripcion'])) {
16 return ["status" => "error", "mensaje" => "Faltan parámetros."];
17 }
18
19 // Buscar dinámicamente el estado "Abierta"
20 $sqlEstado = "SELECT id FROM estado_incidencia WHERE nombre = 'Abierta' LIMIT 1";
21 $resEstado = $conn->query($sqlEstado);
22 $estadoRow = $resEstado ? $resEstado->fetch_assoc() : null;
23 $estadoInicial = $estadoRow ? $estadoRow['id'] : 1;
24
25 // Insertar la nueva incidencia
26 $sql = "INSERT INTO incidencias (id_user, titulo, descripcion, estado_id) VALUES (?, ?, ?, ?)";
27 $stmt = $conn->prepare($sql);
28 $stmt->bind_param("issi", $data['id_user'], $data['titulo'], $data['descripcion'], $estadoInicial);
29
30 if ($stmt->execute()) {
31 return [
32 "status" => "ok",
33 "id_incidencia" => $conn->insert_id,
34 "mensaje" => "Incidencia creada correctamente con estado inicial 'Abierta'."
35 ];
36 } else {
37 return [
38 "status" => "error",
39 "mensaje" => "Error al registrar incidencia: " . $conn->error
40 ];
41 }
42}
