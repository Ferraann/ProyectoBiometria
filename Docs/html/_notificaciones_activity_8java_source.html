46 public void crearCanal(){
47 if(Build.VERSION.SDK_INT >= Build.VERSION_CODES.O){
48 NotificationChannel channel = new NotificationChannel(MY_CHANNEL_ID, "MyCanal", NotificationManager.IMPORTANCE_DEFAULT);
49 channel.enableVibration(true);
50 channel.setVibrationPattern(new long[]{0, 500, 500, 500});
51 NotificationManager notificationManager = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE);
52 notificationManager.createNotificationChannel(channel);
53 }
54 }
55 public void crearNotificacionPeligroExtremo(View v){
56 Intent intent = new Intent(this, NotificacionesActivity.class);
57 intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK );
58 intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TASK);
59
60 int flag = 0;
61 if(Build.VERSION.SDK_INT >= Build.VERSION_CODES.M){
62 flag = PendingIntent.FLAG_IMMUTABLE;
63 }
64
65 PendingIntent pendingIntent = PendingIntent.getActivity(this,0,intent,flag);
66
67 NotificationCompat.Builder builder = new NotificationCompat.Builder(this, MY_CHANNEL_ID)
68// .setSmallIcon(R.drawable.logo_aither)
69 .setContentTitle("Peligro")
70 .setContentIntent(pendingIntent)
71 .setContentText("Cuidado estás es una zona demasiada contaminada")
72 .setPriority(NotificationCompat.PRIORITY_DEFAULT);
73
74
75 if (ActivityCompat.checkSelfPermission(this, android.Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
76 return;
77 }
78
79 NotificationManagerCompat.from(this).notify(1,builder.build());
80
81 Vibrator vibrator = (Vibrator) getSystemService(Context.VIBRATOR_SERVICE);
82 long[] pattern = new long[]{0, 500, 500, 500}; // patrón de vibración
83
84 if (vibrator != null) {
85 if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
86 VibrationEffect effect = VibrationEffect.createWaveform(pattern, 0);
87 vibrator.vibrate(effect);
88 } else {
89 vibrator.vibrate(pattern, 0);
90 }
91
92 long tiempoVibrandoMs = 10000; // 10 segundos, ajusta a lo que quieras
93
94 new Handler(Looper.getMainLooper()).postDelayed(new Runnable() {
95 @Override
96 public void run() {
97 vibrator.cancel();
98 }
99 }, tiempoVibrandoMs);
100 }
101 }
