123 public void botonGuardarModificaciones(View v){
124 SharedPreferences prefs = getSharedPreferences("SesionUsuario", MODE_PRIVATE);
125 String idUsuario = prefs.getString("id", "");
126
127 usuario.setId(idUsuario);
128 usuario.setAction("modificarDatos");
129
130 // -------------------------------------
131 // VALIDAR Y AÑADIR NOMBRE
132 // -------------------------------------
133 if (modificaNombre) {
134 String nuevoNombre = nombre.getText().toString().trim();
135 if (nuevoNombre.isEmpty()) {
136 Toast.makeText(this, "El nombre no puede estar vacío", Toast.LENGTH_SHORT).show();
137 return;
138 }
139 usuario.setNombre(nuevoNombre);
140 }
141
142 // -------------------------------------
143 // VALIDAR Y AÑADIR APELLIDOS
144 // -------------------------------------
145 if (modificaApellido) {
146 String nuevosApellidos = apellidos.getText().toString().trim();
147 if (nuevosApellidos.isEmpty()) {
148 Toast.makeText(this, "Los apellidos no pueden estar vacíos", Toast.LENGTH_SHORT).show();
149 return;
150 }
151 usuario.setApellidos(nuevosApellidos);
152 }
153
154 // -------------------------------------
155 // VALIDAR Y AÑADIR CORREO
156 // -------------------------------------
157 if (modificaCorreo) {
158
159 String correo1 = correoNuevo.getText().toString().trim();
160 String correo2 = repetirCorreoNuevo.getText().toString().trim();
161
162 if (!correo1.equals(correo2)) {
163 Toast.makeText(this, "Los correos no coinciden", Toast.LENGTH_SHORT).show();
164 return;
165 }
166
167
168 usuario.setCorreo(correo1);
169 }
170
171 // -------------------------------------
172 // VALIDAR Y AÑADIR CONTRASEÑA
173 // -------------------------------------
174 if (modificaContrasenya) {
175
176 String pass1 = contrasenyaNueva.getText().toString().trim();
177 String pass2 = repetirContrasenyaNueva.getText().toString().trim();
178
179 if (!pass1.matches(regexTieneMasDe8Caracteres)||!pass1.matches(regexTieneNumeros)||!pass1.matches(regexTieneMayuscula)||!pass1.matches(regexTieneSimbologia)){
180 Toast.makeText(this, "Por favor, introduce una contraseña segura", Toast.LENGTH_SHORT).show();
181 return;
182 }
183 if (!pass1.equals(pass2)) {
184 Toast.makeText(this, "Las contraseñas no coinciden", Toast.LENGTH_SHORT).show();
185 return;
186 }
187
188 usuario.setContrasenya(pass1);
189 }

191 // -------------------------------------

194 putModificarDatos(usuario,this);
