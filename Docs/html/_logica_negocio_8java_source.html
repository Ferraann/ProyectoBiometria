109 call.enqueue(new Callback<PojoRespuestaServidor>() {

111 public void onResponse(Call<PojoRespuestaServidor> call, Response<PojoRespuestaServidor> response) {

113 Log.d("Login", "Error en la respuesta: " + response.code());

115

118
119 // Miramos el status que viene del servidor
120 if (!"ok".equalsIgnoreCase(respuesta.getStatus())) {

122 Log.d("Login", "Login fallido: " + respuesta.getMensaje());
123 return;
124 }
125 PojoUsuario usuarioServidor = respuesta.getUsuario();
126 // 4.4 Guardamos los datos del usuario en SharedPreferences

136 contexto.startActivity(intent);
