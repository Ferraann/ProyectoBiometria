1<?php
2// ------------------------------------------------------------------
3// Fichero: loginUsuario.php
4// Autor: Manuel

6// Descripción:
7// Autentica un usuario verificando sus credenciales
8// (correo y contraseña) contra la base de datos,
9// valida que la cuenta esté activa y establece la
10// sesión si el login es exitoso.
11// ------------------------------------------------------------------
12
13function loginUsuario($conn, $gmail, $password)
14{
15 // Inicia la sesión solo si no está ya iniciada
16 if (session_status() === PHP_SESSION_NONE) {
17 session_start();
18 }
19
20 /* 1. Obtener datos del usuario */
21 $stmt = $conn->prepare(
22 "SELECT id, nombre, apellidos, gmail, password, activo
23 FROM usuario
24 WHERE gmail = ?"
25 );
26
27 if (!$stmt) {
28 return ["status" => "error", "mensaje" => "Error en la base de datos"];
29 }
30
31 $stmt->bind_param("s", $gmail);
32 $stmt->execute();
33 $res = $stmt->get_result();
34
35 if ($res->num_rows === 0) {
36 return ["status" => "error", "mensaje" => "Usuario no encontrado"];
37 }
38
39 $user = $res->fetch_assoc();
40 $stmt->close();
41
42 /* 2. Verificar contraseña */
43 if (!password_verify($password, $user['password'])) {
44 return ["status" => "error", "mensaje" => "Contraseña incorrecta"];
45 }
46
47 /* 3. Comprobar si la cuenta está activa */
48 if (!$user['activo']) {
49 return ["status" => "error", "mensaje" => "Cuenta no activada"];
50 }
51
52 /* 4. Establecer datos de sesión (no guardar contraseña en texto plano) */
53 $_SESSION['usuario_id'] = $user['id'];
54 $_SESSION['usuario_nombre'] = $user['nombre'];
55 $_SESSION['usuario_apellidos'] = $user['apellidos'];
56 $_SESSION['usuario_correo'] = $user['gmail'];
57
58 unset($user['password']);
59 return ["status" => "ok", "usuario" => $user];
60}
