1<?php
2// ------------------------------------------------------------------
3// Fichero: registrarUsuario.php
4// Autor: Manuel
5// Fecha: 5/12/2025
6// ------------------------------------------------------------------
7// Descripción:
8// Función responsable del registro de usuarios. Valida datos,
9// verifica el correo, genera token y envía email de activación.
10// ------------------------------------------------------------------
11
12require_once __DIR__ . '/mailer.php';
13use PHPMailer\PHPMailer\PHPMailer;
14use PHPMailer\PHPMailer\Exception;
15
16function registrarUsuario($conn, $data)
17{
18 /* ---------- 1. Validaciones y saneado ---------- */
19 if (!isset($data['nombre'], $data['apellidos'], $data['gmail'], $data['password'])) {
20 return ["status" => "error", "mensaje" => "Faltan datos obligatorios para el registro."];
21 }
22
23 $nombre = trim($data['nombre']);
24 $apellidos = trim($data['apellidos']);
25 $gmail = trim($data['gmail']);
26 $password = trim($data['password']);
27
28 // Validar formato de correo
29 if (!filter_var($gmail, FILTER_VALIDATE_EMAIL)) {
30 return ["status" => "error", "mensaje" => "El formato del correo electrónico no es válido."];
31 }
32
33 $hash = password_hash($password, PASSWORD_DEFAULT);
34
35 /* ---------- 2. ¿Existe el correo? ---------- */
36 $stmt = $conn->prepare("SELECT id, activo FROM usuario WHERE gmail = ?");
37 $stmt->bind_param("s", $gmail);
38 $stmt->execute();
39 $res = $stmt->get_result();
40
41 if ($res->num_rows) {
42 $usr = $res->fetch_assoc();
43
44 if ($usr['activo'] == 1) {
45 return ["status" => "error", "mensaje" => "El correo ya está registrado y activado."];
46 }
47
48 /* usuario inactivo → lo borramos */
49 $del = $conn->prepare("DELETE FROM usuario WHERE id = ?");
50 $del->bind_param("i", $usr['id']);
51 $del->execute();
52 }
53
54 /* ---------- 3. Crear token seguro ---------- */
55 $token = bin2hex(random_bytes(32)); // 64 caracteres
56 $token_expira = date("Y-m-d H:i:s", time() + 900); // 15 min de validez
57
58 /* ---------- 4. Insertar nuevo usuario (inactivo + token + expiración) ---------- */
59 $ins = $conn->prepare("INSERT INTO usuario
60 (nombre, apellidos, gmail, password, activo, token, token_expira)
61 VALUES (?, ?, ?, ?, 0, ?, ?)");
62 $ins->bind_param("ssssss", $nombre, $apellidos, $gmail, $hash, $token, $token_expira);
63
64 if (!$ins->execute()) {
65 return ["status" => "error", "mensaje" => "Error al registrar el usuario: " . $conn->error];
66 }
67
68 /* ---------- 5. Enviar correo de activación ---------- */
69 $enlace = "https://fsanpra.upv.edu.es/src/html/activacion.html?token=" . urlencode($token);
70
71 $asunto = "Activa tu cuenta en AITHER";
72 $cuerpo = "<h2>¡Hola $nombre!</h2>
73 <p>Gracias por registrarte. Pulsa el botón para activar tu cuenta:</p>
74 <p><a href='$enlace' style='background:#007bff;color:white;padding:10px 20px;text-decoration:none;border-radius:4px;'>Activar cuenta</a></p>
75 <p>Si el botón no funciona, copia y pega esta dirección:<br>$enlace</p>";
76
77 $enviado = false;
78
79 try {
80 $mail = new PHPMailer(true);
81 $mail->isSMTP();
82 $mail->Host = 'smtp.gmail.com';
83 $mail->SMTPAuth = true;
84 $mail->Username = 'no.reply.aither@gmail.com';
85 $mail->Password = 'esdf lkoc qprz rkum';
86 $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;
87 $mail->Port = 587;
88
89 $mail->setFrom('noreply@aither.com', 'AITHER');
90 $mail->addAddress($gmail);
91 $mail->isHTML(true);
92 $mail->Subject = $asunto;
93 $mail->Body = $cuerpo;
94
95 $mail->send();
96 $enviado = true;
97
98 } catch (Exception $e) {
99 error_log('PHPMailer error: ' . $mail->ErrorInfo);
100 return ["status" => "ok", "mensaje" => "Usuario registrado correctamente, pero no se pudo enviar el correo de activación. Contacta con soporte."];
101 }
102
103 return ["status" => "ok", "mensaje" => "Usuario registrado correctamente. Revisa tu correo para activarlo."];
104}
105?>
